Connectez QHSE MEGA à votre base de données Supabase

Pourquoi Supabase?
Vos données seront sur un serveur cloud. Vous pouvez vous connecter depuis n'importe quel appareil et retrouver toutes vos informations.
Gratuit jusqu'a 500MB de données.

Etape 1 — Créer votre projet Supabase

  1. Aller sur supabase.com → Start your project
  2. Créer un compte gratuit (GitHub ou email)
  3. Cliquer New project → donner un nom (ex: "qhse-mega")
  4. Choisir West EU ou Singapore (plus proche de la CI)
  5. Définir un mot de passe de base de données → Create new project
  6. Attendre 2 minutes que le projet se crée

Etape 2 — Créer les tables (coller le SQL)

Dans Supabase → SQL Editor → New query → coller ce code → Run:

-- Activer UUID
create extension if not exists "uuid-ossp";

-- Profils utilisateurs
create table if not exists profiles (
  id uuid references auth.users primary key,
  name text,
  company text,
  phone text,
  created_at timestamptz default now()
);

-- Sites
create table if not exists sites (
  id bigserial primary key,
  user_id uuid references auth.users not null,
  name text not null,
  city text,
  country text default 'Cote d Ivoire',
  address text,
  type text,
  created_at timestamptz default now()
);

-- Enregistrements (tous modules)
create table if not exists records (
  id bigserial primary key,
  user_id uuid references auth.users not null,
  module text not null,
  ref text,
  data jsonb not null default '{}',
  created_at timestamptz default now(),
  updated_at timestamptz default now()
);

-- Documents PDF
create table if not exists docs (
  id bigserial primary key,
  user_id uuid references auth.users not null,
  name text,
  ext text,
  size text,
  date text,
  data text,
  created_at timestamptz default now()
);

-- Securite: chaque utilisateur ne voit que ses donnees
alter table profiles enable row level security;
alter table sites enable row level security;
alter table records enable row level security;
alter table docs enable row level security;

create policy "users_own_profiles" on profiles for all using (auth.uid() = id);
create policy "users_own_sites" on sites for all using (auth.uid() = user_id);
create policy "users_own_records" on records for all using (auth.uid() = user_id);
create policy "users_own_docs" on docs for all using (auth.uid() = user_id);

-- Auto-creer profil a l inscription
create or replace function handle_new_user()
returns trigger as $$
begin
  insert into profiles (id, name, company)
  values (new.id, new.raw_user_meta_data->>'name', new.raw_user_meta_data->>'company');
  return new;
end;
$$ language plpgsql security definer;

create trigger on_auth_user_created
  after insert on auth.users
  for each row execute procedure handle_new_user();

-- IMPORTANT: Dans Supabase > Authentication > Settings
-- Desactivez "Enable email confirmations" pour tester sans confirmation email
-- Ou activez "Confirm email" seulement en production
